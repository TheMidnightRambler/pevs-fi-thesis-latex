\newacronymstyle{simple-desc}
{%
    \GlsUseAcrEntryDispStyle{long-short}%
}
{%
    \renewcommand*{\glossentry}[2]{%
        \item[\glsentryitem{##1}%
        \glstarget{##1}{\glsentryshort{##1}}]
        \glsentrylong{##1}\ifglshasfield{description}{##1}{\newline\glsentrydesc{##1}}{}%
    }%
}

\newacronymstyle{tabular-style}
{%
    \GlsUseAcrEntryDispStyle{long-short}%
}
{%
    \renewcommand*{\glossentry}[2]{%
        \item[]%
        \begin{tabular}{@{}ll@{}}
            \textbf{\glsentryshort{##1}} & \glsentrylong{##1} \\
            & \glsentrydesc{##1}
        \end{tabular}\\[1ex]%
    }%
}
